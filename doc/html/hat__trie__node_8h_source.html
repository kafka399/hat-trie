00080     // accessors

00082         if (*p == '\0') {
00083             return word();
00084         }
00085         return _store.exists(p);
00086     }
00087     size_t size() const { return _store.size(); }
00088     bool word() const { return _word; }
00089 
00090     // modifiers
00091     bool insert(const char *p) {
00092         if (*p == '\0') {
00093             bool b = word();
00094             set_word(true);
00095             return !b;
00096         }
00097         return _store.insert(p);
00098     }
00099     void set_word(bool b) { _word = b; }

00109 class hat_trie_node : public hat_trie_node_base {
00110     friend class hat_trie<std::string>;

00112   private:
00113     typedef hat_trie_node_base _node_base;
00114 
00115   public:
00116     hat_trie_node(char ch = '\0') : _node_base(ch) {
00117         for (int i = 0; i < HT_ALPHABET_SIZE; ++i) {
00118             _children[i] = NULL;
00119         }
00120         set_word(false);
00121     }
00122     ~hat_trie_node() {
00123         for (int i = 0; i < HT_ALPHABET_SIZE; ++i) {
00124             delete _children[i];
00125         }
00126     }
00127 
00128     // accessors
00129     bool word() const { return _types[HT_ALPHABET_SIZE]; }
00130 
00131     // modifiers
00132     void set_word(bool b) { _types[HT_ALPHABET_SIZE] = b; }
00133 
00134   private:
00135     std::bitset<HT_ALPHABET_SIZE + 1> _types;  // extra bit is an end of word flag
00136     _node_base *_children[HT_ALPHABET_SIZE];  // untyped pointers to children
